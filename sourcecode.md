def unzip(file, target):
    with zipfile.ZipFile(file, "r") as zipref:
        zipref.extractall(target)
        return True

def hashcheck(target):
    with open(target, "rb") as f:
        data = f.read()
        sha = hashlib.sha256(data).hexdigest()
    return sha

def curl(file, url, timeout = None):
    try:
        with httpx.Client(headers = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36 SeecretMC/1.3"}) as client:
            r = client.get(url)
            r.raise_for_status()
            with open(file, "wb") as f:
                f.write(r.content)
            return True
    except:
        return False

def curlstreaming(file, url, timeout = None):
    try:
        with httpx.Client(timeout = timeout, headers = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36 SeecretMC/1.3"}) as client:
            with httpx.stream("GET", url) as r:
                r.raise_for_status()

                with open(file, "wb") as f:
                    for chunk in r.iter_bytes(chunk_size = 16384):
                        f.write(chunk)
        return True
    except:
        return False
